            // 4. - 로 쪼개지는 숫자가 총 3개고, 3, 3, 4의 길이가 각각 맞나?
            for (let i=1; i<phone_info.length; i++) {
                if (phone_info[i].length != 4) {
                    return false;
                }
            }
